```sql
-- This query calculates the average order value (AOV) for each product category.
-- It joins the orders, order_items, and products tables to associate order items with their respective product categories.
-- The total revenue for each order is calculated first, and then the average of these total revenues is computed per category.

SELECT
    p.category,
    AVG(order_total_revenue.total_revenue) AS average_order_value
FROM
    `bigquery-public-data.thelook_ecommerce.products` AS p
JOIN
    `bigquery-public-data.thelook_ecommerce.order_items` AS oi
    ON p.product_id = oi.product_id
JOIN
    (
        -- Subquery to calculate the total revenue for each order.
        -- This pattern is similar to Example 1, which calculates total revenue per order.
        SELECT
            o.order_id,
            SUM(oi_inner.sale_price) AS total_revenue
        FROM
            `bigquery-public-data.thelook_ecommerce.orders` AS o
        JOIN
            `bigquery-public-data.thelook_ecommerce.order_items` AS oi_inner
            ON oi_inner.order_id = o.order_id
        GROUP BY
            o.order_id
    ) AS order_total_revenue
    ON oi.order_id = order_total_revenue.order_id
GROUP BY
    p.category
ORDER BY
    average_order_value DESC;
```
