-- Calculate the conversion rate (orders per user) for the past 6 months.
SELECT
    CAST(COUNT(DISTINCT o.order_id) AS BIGNUMERIC) / COUNT(DISTINCT u.id) AS conversion_rate
FROM
    `your_project.your_dataset.users` AS u
LEFT JOIN
    `your_project.your_dataset.orders` AS o
ON
    u.id = o.user_id
WHERE
    -- Filter orders to include only those placed within the last 6 months.

    -- have orders in the specified period.
