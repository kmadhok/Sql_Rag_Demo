```sql
-- This query identifies the top 10 users based on their total lifetime revenue.
-- It joins the 'users' and 'order_items' tables to aggregate sale prices per user.
SELECT
    oi.user_id,
    SUM(oi.sale_price) AS total_revenue
FROM
    `mock-project.mock_dataset.users` AS u
JOIN
    `mock-project.mock_dataset.order_items` AS oi
ON
    u.id = oi.user_id
GROUP BY
    oi.user_id
ORDER BY
    total_revenue DESC
LIMIT 10;
```
